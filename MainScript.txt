btns:Button(
    "Player Esp",
    function()
        local function createBillboard(entity)


    local billboardGui = Instance.new("BillboardGui")
    billboardGui.Name = "BillboardGui"
    billboardGui.AlwaysOnTop = true
    billboardGui.Size = UDim2.new(1, 0, 1, 0)
    billboardGui.StudsOffset = Vector3.new(0, 5, 0)
    billboardGui.Adornee = entity
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(5.5, 0, 5.5, 0) -- Adjust the values to make the circle bigger
    frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    frame.BackgroundTransparency = 0.5
    frame.BorderSizePixel = 0
    frame.ClipsDescendants = true
    frame.Parent = billboardGui
    
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0.5, 0)
    corner.Parent = frame
    
    billboardGui.Parent = entity
end

local function checkEntities()
    local entities = game:GetService("Workspace").Entities:GetDescendants()
    
    for _, entity in ipairs(entities) do
        if entity:IsA("Model") then
            local characterHandler = entity:FindFirstChild("CharacterHandler")
            
            if characterHandler then
                createBillboard(entity)
            end
        end
    end
end

game:GetService("Players").PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        wait(5) -- Delay to ensure the character is fully loaded
        checkEntities()
    end)
end)
    end
)
